create table if not exists public.students
(
    enrolmentno  integer                              not null,
    studentname  varchar(50)                          not null,
    dob          date                                 not null,
    branch       varchar(5)                           not null,
    studentclass varchar(3)                           not null,
    subjectcodes text[],
    sessionid    text default NULL::character varying not null,
    primary key (enrolmentno, sessionid)
);

alter table public.students
    owner to postgres;
